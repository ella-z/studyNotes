# 继承
- 继承是一种关系，是父类级别与类级别的关系。
```
🌰:
function Person(name,age,sex){
  this.name = name;
  this.age = age;
  this.sex = sex;
}
Person.prototype.eat = function(){
  console.log("吃东西");
}
function Student(score){
  this.score = score;
}
//改变学生原型的指向=》学生继承了Person
Student.prototype = new Person(Jack,18,man);
Student.prototype.study = function(){
  console.log("study");
}

var stu1 = new Student(100);
console.log(stu1.name);//Jack
stu.study();//100

//但是通过改变原形指向实现继承，会有缺陷：因为改变原型指向的同时实现继承，直接初始化了属性，继承过来的属性的值都是一样的，不可改变。
var stu1 = new Student(100);
console.log(stu1.name);//Jack
var stu2 = new Student(60);
console.log(stu2.name);//Jack
//只能重新用属性的值来改变
stu2.name = "John";
console.log(stu2.name);//John

//解决方案：继承的时候不改变原型的指向，直接调用父级的构造函数的方式来为属性赋值就可以了，即为借用构造函数:构造函数名字.call(当前对象，属性，属性，属性)。
//缺陷：父级类别的方法没有被继承。
function Person(name,age,sex){
  this.name = name;
  this.age = age;
  this.sex = sex;
}
Person.prototype.eat = function(){
  console.log("吃东西");
}
function Student(name,age,sex,score){
  Person.call(this,name,age,sex,weight);
  this.score = score;
}
Student.prototype.study = function(){
  console.log("study");
}
var stu1 = new Student("Jack",18,"man",100);
console.log(stu1.name);//Jack
var stu2 = new Student("john",18,"man",100);
console.log(stu2.name);//john
stu1.eat();//error
```

### 组合继承
- 组合继承=原型继承+借用构造函数继承
```
🌰:
function Person(name,age,sex){
  this.name = name;
  this.age = age;
  this.sex = sex;
}
Person.prototype.eat = function(){
  console.log("吃东西");
}
function Student(name,age,sex,score){
 Person.call(this,name,age,sex,weight);
  this.score = score;
}
//改变学生原型的指向=》学生继承了Person
Student.prototype = new Person();
Student.prototype.study = function(){
  console.log("study");
}
var stu1 = new Student("Jack",18,"man",100);
console.log(stu1.name);//Jack
var stu2 = new Student("john",18,"man",100);
console.log(stu2.name);//john
stu1.eat();//吃东西

//属性与方法都被继承了
```

### 拷贝继承
- 拷贝继承：把一个对象中的属性或者方法通过遍历直接复制到另一个对象中。
```
🌰:
//简单的拷贝继承
var obj1 ={
 name:"jack",
 age:18,
 eat:function(){
    console.log("eating~");
 }
};

//仅仅改变了地址的指向
var obj2 = obj1;
console.log(obj2.name);//Jack
obj2.eat();//eating~

//拷贝继承
function Person(age,sex){
  this.age = age;
  this.sex = sex;
}
Person.prototype.name = "Jack";
Person.prototype.eat = function(){
  console.log("吃东西");
}

var obj2={};

for(var key in Person.prototype){
  obj2[key] = Person.prototype[key];
}

console.log(obj2.name);//Jack

```
