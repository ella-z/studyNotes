# 闭包
- 闭包指有权访问另一个函数作用域中的变量的函数。
- 闭包的作用：缓存数据，延长作用域链
- 闭包的优点和缺点：缺点：没能及时地释放内存，优点：缓存数据
- 闭包的模式：分为函数闭包和对象闭包
- 闭包的应用

- 局部变量是在函数中，函数使用结束后，局部变量就会被自动释放。但加入闭包后里面的局部变量的使用作用域链就会被延长。

```
🌰：

//函数模式的闭包：在一个函数中包含着另一个函数
function f1(){
  var num = 10;
  function f2(){
    console.log(num);
  }
  f2();
}
f1();

//对象模式闭包：在函数中包含着一个对象
function f3(){
  var num = 10;
  var obj={
    age:num
  };
  console.log(obj.num);
}
f3();

//闭包
function f1(){
  var num = 10;
  return function(){
    console.log(num);
    return num;
  }
}
var ff = f1();//获取返回值
var result = ff();
console.log(result);//10

function f2(){
  var num = 10;
  return {
    age:num;
  }
}

var obj = f2();
console.log(obj.age);

//闭包小案例
function f1(){
  var num = 10;
  num++;
  return num;
}
f1();//11
f1();//11
f1();//11

function f2(){
  var num = 10;
  retrun function(){
    num++;
    return num;
  }
}
var ff = f2();
ff();//11
ff();//12
ff();//13
//因为有数据缓存

//闭包方式，产生三个随机数,但都是相同的
function f1(){
  var num = parseInt(Math.random()*10+1);
  return function(){
    console.log(num);
  }
}
var ff = f1();
ff();
ff();
//这两次输出的结果都一致
```

###区分是否为闭包的小例子
```
🌰1：
var name = "window";
var obj = {
  name:"object",
  getName: function(){
    return function(){
       return this.name;
  }  
 }
}

console.log(obj.getName()());//window

//分析
//首先obj调用了getName函数，然后getName函数返回了一个函数使用f接收。
var f = obj.getName();
//然后返回的函数再自调用
f();//所以此时的this指向的是window而并非obj，所以函数中并无局部变量则没有闭包产生

🌰2：
var name = "window";
var obj = {
  name:"object",
  getName: function(){
    var that = this;
    return function(){
       return that.name;
  }  
 }
}

console.log(obj.getName()());//object



```
